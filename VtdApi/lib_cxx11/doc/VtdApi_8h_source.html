361 void VtdApi::registerEventFunc(ReceiverT* receiver, void (ReceiverT::*memberFunc)(const EventT*), const ConnectionHandle& connectionHandle)
362 {
363  MessageHandler messageHandle;
364 
365  switch (EventT::MsgId)
366  {
367  case MESSAGE_RDB:
368  {
369  messageHandle.setHandler<EventT>(boost::bind(memberFunc, receiver, boost::placeholders::_1));
370  connectServiceMap_.find(connectionHandle)->second->handleRDBMessages_.insert(std::make_pair(connectionHandle, messageHandle));
371  }break;
372 
373  case MESSAGE_SCP:
374  {
375  messageHandle.setHandler<EventT>(boost::bind(memberFunc, receiver, boost::placeholders::_1));
376  connectServiceMap_.find(connectionHandle)->second->callbackScp_.push_back(messageHandle);
377  }break;
378 
379  default:
380  break;
381  }
382 }
